140 void QextSerialPortPrivate::setParity(ParityType parity, bool update)
141 {

144  if (settings.DataBits == DATA_8) {
145 #ifdef Q_OS_WIN

152

259  if (settings.DataBits == DATA_5) {
260  QESP_WARNING("QextSerialPort: 2 stop bits cannot be used with 5 data bits");
261  }
262  else {
263  settings.StopBits = stopBits;
264  settingsDirtyFlags |= DFE_StopBits;
265  }
266  break;
267  default:
268  QESP_WARNING()<<"QextSerialPort does not support stop bits: "<<stopBits;
269  }
270  if (update && q_func()->isOpen())

304 void QextSerialPortPrivate::_q_canRead()
305 {
306  qint64 maxSize = bytesAvailable_sys();
307  if (maxSize > 0) {
308  char *writePtr = readBuffer.reserve(size_t(maxSize));
309  qint64 bytesRead = readData_sys(writePtr, maxSize);
310  if (bytesRead < maxSize)
311  readBuffer.chop(maxSize - bytesRead);
312  Q_Q(QextSerialPort);
313  Q_EMIT q->readyRead();
314  }
315 }

717 QString QextSerialPort::errorString()

724  case E_INVALID_FD:

728  case E_CAUGHT_NON_BLOCKED_SIGNAL:
729  return tr("Caught a non-blocked signal (POSIX)");
730  case E_PORT_TIMEOUT:
731  return tr("Operation timed out (POSIX)");
732  case E_INVALID_DEVICE:
733  return tr("The file opened by the port is not a valid device");
734  case E_BREAK_CONDITION:
735  return tr("The port detected a break condition");
736  case E_FRAMING_ERROR:
737  return tr("The port detected a framing error (usually caused by incorrect baud rate settings)");
738  case E_IO_ERROR:
739  return tr("There was an I/O error while communicating with the port");
740  case E_BUFFER_OVERRUN:
741  return tr("Character buffer overrun");
742  case E_RECEIVE_OVERFLOW:
743  return tr("Receive buffer overflow");
744  case E_RECEIVE_PARITY_ERROR:
745  return tr("The port detected a parity error in the received data");
746  case E_TRANSMIT_OVERFLOW:
747  return tr("Transmit buffer overflow");
748  case E_READ_FAILED:
749  return tr("General read operation failure");
750  case E_WRITE_FAILED:
751  return tr("General write operation failure");

979 qint64 QextSerialPort::readData(char *data, qint64 maxSize)
980 {
981  Q_D(QextSerialPort);
982  QWriteLocker locker(&d->lock);
983  qint64 bytesFromBuffer = 0;
984  if (!d->readBuffer.isEmpty()) {
985  bytesFromBuffer = d->readBuffer.read(data, maxSize);
986  if (bytesFromBuffer == maxSize)
987  return bytesFromBuffer;
988  }
989  qint64 bytesFromDevice = d->readData_sys(data+bytesFromBuffer, maxSize-bytesFromBuffer);
990  if (bytesFromDevice < 0) {
991  return -1;
992  }
993  return bytesFromBuffer + bytesFromDevice;
994 }
