00048     \param  void
00049     \return void
00050 */
00051 void InitPLL (void)

00059 
00060     PCLKSEL0  = PCLKSEL0_Value;     /* Peripheral Clock Selection         */
00061     PCLKSEL1  = PCLKSEL1_Value;
00062

00064 
00065     PLL0CFG   = PLL0CFG_Value;      /* configure PLL0                     */
00066     PLL0FEED  = 0xAA;
00067     PLL0FEED  = 0x55;
00068 
00069     PLL0CON   = 0x01;             /* PLL0 Enable                        */
00070     PLL0FEED  = 0xAA;
